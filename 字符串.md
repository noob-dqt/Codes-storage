```
vector<int>hs(n+1,0),bs(n+1,1);
for(int i=1;i<=n;i++){
    hs[i]=(1LL*hs[i-1]*base+s[i-1]-'a'+1)%mod;
    bs[i]=1LL*bs[i-1]*base%mod; //0, base, base^2, ...
}
auto geths=[&](int l,int r){ //字符串l,r下标内子串哈希值，0<=l,r<n
    return (1LL*hs[r+1]-1LL*hs[l]*bs[r-l+1]%mod+mod)%mod;
};
```
